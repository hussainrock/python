import seaborn as sns
import matplotlib.pyplot as plt

    'ParentEduc': ["associate's degree", "bachelor's degree", "high school", "master's degree", "some college", "some high school"],

    'WritingScore': [70.299099, 73.331069, 65.421136, 76.356896, 68.501432, 63.632409]
}

sns.barplot(x='ParentEduc', y='WritingScore', data=df2, color='lightcoral', label='Writing Score')

plt.xlabel("Parent Education Level")
plt.ylabel("Scores")
plt.title("Scores by Parent Education Level")
plt.xticks(rotation=45)

plt.show()
